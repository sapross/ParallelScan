\begin{frame}{Vectorization}
	Requirements:
	\begin{itemize}
		\item No loop carried dependency
		\item Loop bounds
		\item No jumps in code
	\end{itemize}
	
	Realising it:
	\begin{itemize}
		\item \#pragma omp simd
		\item Compiling with -O3
		\item Using Intel Icx Compiler
	\end{itemize}
\end{frame}
